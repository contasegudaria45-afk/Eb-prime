local function ColorToTable(col)
    return { math.floor(col.R * 255 + 0.5), math.floor(col.G * 255 + 0.5), math.floor(col.B * 255 + 0.5) }
end

local function TableToColor(t, fallback)
    if type(t) == "table" and tonumber(t[1]) and tonumber(t[2]) and tonumber(t[3]) then
        return Color3.fromRGB(math.clamp(tonumber(t[1]),0,255), math.clamp(tonumber(t[2]),0,255), math.clamp(tonumber(t[3]),0,255))
    end
    return fallback
end

local function SaveUISettings()
    if not writefile then return end
    local out = {
        Transparency = UISettings.Transparency or 0,
        LightMode = UISettings.LightMode or false,
        TAFDivision = UISettings.TAFDivision or "Exibir Todas",
        AimbotTarget = UISettings.AimbotTarget or "Head",
        Theme = {}
    }
    for k,v in pairs(C) do
        if typeof(v) == "Color3" then out.Theme[k] = ColorToTable(v) end
    end
    pcall(function() writefile(CONFIG_ARQUIVO, HttpService:JSONEncode(out)) end)
end

local function LoadUISettings()
    if not (readfile and isfile and isfile(CONFIG_ARQUIVO)) then return end
    local ok, raw = pcall(readfile, CONFIG_ARQUIVO)
    if not ok or not raw or raw == "" then return end
    local ok2, data = pcall(function() return HttpService:JSONDecode(raw) end)
    if not ok2 or type(data) ~= "table" then return end
    UISettings.Transparency = tonumber(data.Transparency) or 0
    UISettings.LightMode = data.LightMode == true
    UISettings.TAFDivision = type(data.TAFDivision)=="string" and data.TAFDivision or "Exibir Todas"
    UISettings.AimbotTarget = (data.AimbotTarget == "Chest") and "Chest" or "Head"
    if type(data.Theme) == "table" then
        for k,v in pairs(data.Theme) do
            if C[k] and typeof(C[k]) == "Color3" then
                C[k] = TableToColor(v, C[k])
            end
        end
    end
end

-- ══════════════════════════════════════════════
--  HTTP COM RETRY
-- ══════════════════════════════════════════════
local function HttpGetRetry(url, tentativas)
    tentativas = tentativas or 3
    for i = 1, tentativas do
        local ok, res = pcall(game.HttpGet, game, url)
        if ok and res and #res > 5 then return true, res end
        if i < tentativas then task.wait(1.5) end
    end
    return false, "Falha na conexão após "..tentativas.." tentativas"
end

-- ══════════════════════════════════════════════
--  PARSE JSON v11
-- ══════════════════════════════════════════════
local function ParseJSON_v11(raw)
    local cb = raw:match('"config"%s*:%s*(%b{})')
    if cb then
        ServerConfig.status         = cb:match('"status"%s*:%s*"([^"]*)"')         or "online"
        ServerConfig.status_message = cb:match('"status_message"%s*:%s*"([^"]*)"') or ""
        ServerConfig.version        = cb:match('"version"%s*:%s*"([^"]*)"')         or "v11.0"
        ServerConfig.update_notes   = cb:match('"update_notes"%s*:%s*"([^"]*)"')   or ""
    end
    local kb = raw:match('"keys"%s*:%s*(%b[])')
    local keys = {}
    if kb then
        for entry in kb:gmatch('%b{}') do
            local k   = entry:match('"key"%s*:%s*"([^"]+)"')
            local exp = entry:match('"expires"%s*:%s*(%d+)')
            local bnd = entry:match('"bound_user"%s*:%s*"([^"]*)"')
            if k then
                table.insert(keys,{
                    key        = k,
                    expires    = tonumber(exp),
                    bound_user = (bnd and bnd ~= "" and bnd or nil)
                })
            end
        end
    end
    return keys
end

-- ══════════════════════════════════════════════
--  CHECK KEY v11
-- ══════════════════════════════════════════════
local function CheckKey(userKey)
    local ok, raw = HttpGetRetry(KEY_URL, 3)
    if not ok then return false, "🔌 "..raw end

    local keys = ParseJSON_v11(raw)

    local st = ServerConfig.status
    if st == "offline" then
        local m = ServerConfig.status_message ~= "" and ServerConfig.status_message or "Servidor offline."
        return false, "🔴 OFFLINE: "..m
    elseif st == "maintenance" then
        local m = ServerConfig.status_message ~= "" and ServerConfig.status_message or "Em manutenção."
        return false, "🟣 MANUTENÇÃO: "..m
    elseif st == "updating" then
        local n = ServerConfig.update_notes ~= "" and ServerConfig.update_notes or "Atualização em andamento."
        return false, "🟡 ATUALIZANDO: "..n
    end

    for _, entry in ipairs(keys) do
        if entry.key == userKey then
            if entry.bound_user then
                if tostring(Player.UserId) ~= entry.bound_user then
                    return false, "🔒 Key vinculada a outro jogador!"
                end
            end
            if entry.expires then
                if os.time() > entry.expires then
                    return false, "⌛ Key expirada!"
                end
                return true, entry.expires
            end
            return true, 9999999999
        end
    end
    return false, "❌ Key inválida ou não encontrada!"
end

-- ══════════════════════════════════════════════
--  DRAG HELPER
-- ══════════════════════════════════════════════
local function MakeDraggable(frame)
    local drag=false; local dInp=nil; local dStart=nil; local sPos=nil
    frame.InputBegan:Connect(function(i)
        if isIn(i) then
            drag=true; dStart=i.Position; sPos=frame.Position
            i.Changed:Connect(function()
                if i.UserInputState==Enum.UserInputState.End then drag=false end
            end)
        end
    end)
    frame.InputChanged:Connect(function(i) if isDr(i) then dInp=i end end)
    UIS.InputChanged:Connect(function(i)
        if i==dInp and drag then
            local d=i.Position-dStart
            frame.Position=UDim2.new(sPos.X.Scale,sPos.X.Offset+d.X,sPos.Y.Scale,sPos.Y.Offset+d.Y)
        end
    end)
end

    local function SegmentedSelect(parent,label,options,default,onChange,order,columns)
        -- Novo sistema de seleção: sem dropdown flutuante, sem clipping e sem conflito de ZIndex.
        -- Funciona como botões segmentados, com área de toque grande e callback instantâneo.
        local cur = default or options[1]
        local cols = math.clamp(columns or math.min(#options,3), 1, 4)
        local rows = math.ceil(#options / cols)
        local maxRows = 4
        local visibleRows = math.min(rows, maxRows)
        local cellH, gap = 34, 6
        local gridH = visibleRows * cellH + math.max(visibleRows-1,0) * gap
        local canvasH = rows * cellH + math.max(rows-1,0) * gap
        local totalH = 42 + gridH + 12

        local wrap=Instance.new("Frame"); wrap.LayoutOrder=order or 5
        wrap.Size=UDim2.new(1,0,0,totalH); wrap.BackgroundColor3=C.BGT
        wrap.ZIndex=25; wrap.Parent=parent; wrap.Active=true
        Rnd(wrap,9); Stk(wrap,1,C.A,0.85)

        local lbl=Instance.new("TextLabel"); lbl.Size=UDim2.new(0.58,-12,0,24)
        lbl.Position=UDim2.new(0,12,0,8); lbl.BackgroundTransparency=1
        lbl.Text=label; lbl.TextColor3=C.Tx; lbl.Font=Enum.Font.Gotham
        lbl.TextSize=11; lbl.TextXAlignment=Enum.TextXAlignment.Left; lbl.ZIndex=26; lbl.Parent=wrap

        local vl=Instance.new("TextLabel"); vl.Size=UDim2.new(0.42,-12,0,24)
        vl.Position=UDim2.new(0.58,0,0,8); vl.BackgroundTransparency=1
        vl.Text=tostring(cur); vl.TextColor3=C.A; vl.Font=Enum.Font.GothamBold
        vl.TextSize=10; vl.TextXAlignment=Enum.TextXAlignment.Right; vl.ZIndex=26; vl.Parent=wrap

        local grid=Instance.new(rows > maxRows and "ScrollingFrame" or "Frame")
        grid.Size=UDim2.new(1,-24,0,gridH)
        grid.Position=UDim2.new(0,12,0,38); grid.BackgroundTransparency=1
        grid.ZIndex=26; grid.Parent=wrap; grid.Active=true
        if grid:IsA("ScrollingFrame") then
            grid.BorderSizePixel=0
            grid.ScrollBarThickness=2
            grid.ScrollBarImageColor3=C.A
            grid.CanvasSize=UDim2.new(0,0,0,canvasH)
        end

        local layout=Instance.new("UIGridLayout")
        layout.SortOrder=Enum.SortOrder.LayoutOrder
        layout.FillDirection=Enum.FillDirection.Horizontal
        layout.HorizontalAlignment=Enum.HorizontalAlignment.Center
        layout.VerticalAlignment=Enum.VerticalAlignment.Top
        layout.CellPadding=UDim2.new(0,gap,0,gap)
        layout.CellSize=UDim2.new(1/cols, -gap, 0, cellH)
        layout.Parent=grid

        local buttons={}
        local function refresh()
            vl.Text=tostring(cur)
            for _,b in ipairs(buttons) do
                local active = b.Value == cur
                b.Button.BackgroundColor3 = active and C.A or C.Input
                b.Button.TextColor3 = active and C.Black or C.Tx
                b.Button.TextTransparency = active and 0 or 0.08
                local st=b.Button:FindFirstChildWhichIsA("UIStroke")
                if st then
                    st.Color = active and C.Glow or C.A
                    st.Transparency = active and 0.15 or 0.72
                    st.Thickness = active and 1.4 or 1
                end
            end
        end

        for i,opt in ipairs(options) do
            local b=Instance.new("TextButton")
            b.LayoutOrder=i; b.Text=tostring(opt); b.AutoButtonColor=false
            b.BackgroundColor3=C.Input; b.TextColor3=C.Tx; b.Font=Enum.Font.GothamBold
            b.TextSize=10; b.ZIndex=27; b.Parent=grid; b.Active=true
            Rnd(b,8); Stk(b,1,C.A,0.72)
            table.insert(buttons,{Button=b,Value=opt})
            b.MouseEnter:Connect(function()
                if opt ~= cur then Tw(b,Ti(0.12),{BackgroundColor3=C.Card}):Play() end
            end)
            b.MouseLeave:Connect(function()
                if opt ~= cur then Tw(b,Ti(0.12),{BackgroundColor3=C.Input}):Play() end
            end)
            b.MouseButton1Click:Connect(function()
                if cur == opt then
                    refresh()
                    return
                end
                cur=opt
                refresh()
                Tw(b,Ti(0.08),{BackgroundTransparency=0.08}):Play()
                task.delay(0.08,function()
                    if b and b.Parent then Tw(b,Ti(0.10),{BackgroundTransparency=0}):Play() end
                end)
                if onChange then onChange({opt}, opt) end
            end)
        end
        refresh()
        return wrap
    end

    local function sameColor(a,b)
        return typeof(a)=="Color3" and typeof(b)=="Color3"
            and math.abs(a.R-b.R)<0.002 and math.abs(a.G-b.G)<0.002 and math.abs(a.B-b.B)<0.002
    end

    local function mapColor(col, oldC)
        for k,old in pairs(oldC) do
            if sameColor(col, old) and C[k] then return C[k] end
        end
        return col
    end

    local function mapSequence(seq, oldC)
        local pts={}
        for _,kp in ipairs(seq.Keypoints) do
            table.insert(pts, ColorSequenceKeypoint.new(kp.Time, mapColor(kp.Value, oldC)))
        end
        return ColorSequence.new(pts)
    end

    local function RefreshTheme(oldC)
        local roots={ScreenGui}
        for _,root in ipairs(roots) do
            if root and root.Parent then
                for _,obj in ipairs(root:GetDescendants()) do
                    pcall(function()
                        if obj:IsA("GuiObject") then
                            obj.BackgroundColor3 = mapColor(obj.BackgroundColor3, oldC)
                            if obj:IsA("TextLabel") or obj:IsA("TextButton") or obj:IsA("TextBox") then
                                obj.TextColor3 = mapColor(obj.TextColor3, oldC)
                                if obj:IsA("TextBox") then obj.PlaceholderColor3 = mapColor(obj.PlaceholderColor3, oldC) end
                            end
                            if obj:IsA("ScrollingFrame") then obj.ScrollBarImageColor3 = mapColor(obj.ScrollBarImageColor3, oldC) end
                            if obj:IsA("ImageLabel") or obj:IsA("ImageButton") then obj.ImageColor3 = mapColor(obj.ImageColor3, oldC) end
                        elseif obj:IsA("UIStroke") then
                            obj.Color = mapColor(obj.Color, oldC)
                        elseif obj:IsA("UIGradient") then
                            obj.Color = mapSequence(obj.Color, oldC)
                        end
                    end)
                end
            end
        end
    end

    local function ApplyTransparency(v, save)
        v = math.clamp(tonumber(v) or 0, 0, 10)
        UISettings.Transparency = v
        local amount = v / 10
        if Win then
            -- Guarda a transparência original de cada componente para o slider poder ir e voltar sem acumular erro.
            if Win:GetAttribute("DeltaBaseTransparency") == nil then
                Win:SetAttribute("DeltaBaseTransparency", Win.BackgroundTransparency)
            end
            local winBase = Win:GetAttribute("DeltaBaseTransparency") or Win.BackgroundTransparency
            Win.BackgroundTransparency = math.clamp(winBase + (amount * (1 - winBase)), 0, 1)
            for _,obj in ipairs(Win:GetDescendants()) do
                if obj:IsA("GuiObject") then
                    if obj:GetAttribute("DeltaBaseTransparency") == nil then
                        obj:SetAttribute("DeltaBaseTransparency", obj.BackgroundTransparency)
                    end
                    local base = obj:GetAttribute("DeltaBaseTransparency") or obj.BackgroundTransparency
                    if base < 1 then
                        obj.BackgroundTransparency = math.clamp(base + (amount * (1 - base)), 0, 1)
                    end
                elseif obj:IsA("UIStroke") then
                    if obj:GetAttribute("DeltaBaseTransparency") == nil then
                        obj:SetAttribute("DeltaBaseTransparency", obj.Transparency)
                    end
                    local base = obj:GetAttribute("DeltaBaseTransparency") or obj.Transparency
                    obj.Transparency = math.clamp(base + (amount * (1 - base)), 0, 1)
                end
            end
        end
        if save ~= false then SaveUISettings() end
    end

    local function ApplyTheme(newC, save)
        local oldC={}
        for k,v in pairs(C) do oldC[k]=v end
        for k,v in pairs(newC) do
            if C[k] ~= nil and typeof(v)=="Color3" then C[k]=v end
        end
        RefreshTheme(oldC)
        ApplyTransparency(UISettings.Transparency or 0, false)
        if save ~= false then SaveUISettings() end
        Notify("Personalização","Tema aplicado em tempo real! ✓",3,"ok")
    end

local function OtimizarObj(obj)
    if not isOptimized then return end
    if not obj:IsDescendantOf(workspace) then return end
    if Player.Character and obj:IsDescendantOf(Player.Character) then return end
    local ok,cn=pcall(function() return obj.ClassName end); if not ok then return end
    if OriginalStates.Parts[obj] then return end
    local name=obj.Name
    if HIDE_MODEL[name] or name=="Null" then
        OriginalStates.Parts[obj]={_t="model",Parent=obj.Parent}
        pcall(function() obj.Parent=OriginalStates.HiddenObjects end); optimizedCount+=1; return
    end
    if cn=="UnionOperation" then
        OriginalStates.Parts[obj]={_t="union",CastShadow=obj.CastShadow,RenderFidelity=obj.RenderFidelity,Material=obj.Material}
        pcall(function() obj.CastShadow=false; obj.RenderFidelity=Enum.RenderFidelity.Performance
            if extremeMode then obj.Material=Enum.Material.Plastic end end); optimizedCount+=1; return
    end
    if cn=="Texture" or cn=="Decal" then
        OriginalStates.Parts[obj]={_t="decal",Transparency=obj.Transparency}
        pcall(function() obj.Transparency=1 end); optimizedCount+=1; return
    end
    if cn=="SurfaceAppearance" then
        OriginalStates.Parts[obj]={_t="pbr",Parent=obj.Parent}
        pcall(function() obj.Parent=OriginalStates.HiddenObjects end); optimizedCount+=1; return
    end
    if cn=="ParticleEmitter" or cn=="Trail" or cn=="Smoke" or cn=="Fire" or cn=="Sparkles" or cn=="Beam" then
        OriginalStates.Parts[obj]={_t="emitter",Enabled=obj.Enabled}
        pcall(function() obj.Enabled=false end); optimizedCount+=1; return
    end
    if cn=="PointLight" or cn=="SpotLight" or cn=="SurfaceLight" then
        OriginalStates.Parts[obj]={_t="light",Enabled=obj.Enabled,Brightness=obj.Brightness}
        pcall(function() obj.Enabled=false; obj.Brightness=0 end); optimizedCount+=1; return
    end
    if cn=="ProximityPrompt" then
        OriginalStates.Parts[obj]={_t="prompt",MaxActivationDistance=obj.MaxActivationDistance}
        pcall(function() obj.MaxActivationDistance=0 end); optimizedCount+=1; return
    end
    if obj:IsA("BasePart") then
        if PROTECTED[name] then return end
        local e={_t="part",Material=obj.Material,CastShadow=obj.CastShadow,Reflectance=obj.Reflectance}
        if obj:IsA("MeshPart") then e.RenderFidelity=obj.RenderFidelity end
        OriginalStates.Parts[obj]=e
        pcall(function()
            obj.CastShadow=false; obj.Reflectance=0
            obj.Material=extremeMode and Enum.Material.Plastic or Enum.Material.SmoothPlastic
            if obj:IsA("MeshPart") then obj.RenderFidelity=Enum.RenderFidelity.Performance end
        end); optimizedCount+=1
    end
end

function AlternarBooster(v)
    isOptimized=v
    if isOptimized then
        optimizedCount=0
        OriginalStates.Lighting={
            GlobalShadows=Lighting.GlobalShadows,Brightness=Lighting.Brightness,FogEnd=Lighting.FogEnd,
            WaterWaveSize=workspace.Terrain.WaterWaveSize,WaterWaveSpeed=workspace.Terrain.WaterWaveSpeed,
            WaterReflectance=workspace.Terrain.WaterReflectance,
        }
        Lighting.GlobalShadows=false; Lighting.Brightness=1; Lighting.FogEnd=1000000
        workspace.Terrain.WaterWaveSize=0; workspace.Terrain.WaterWaveSpeed=0; workspace.Terrain.WaterReflectance=0
        for _,fx in ipairs(Lighting:GetChildren()) do
            if fx:IsA("PostEffect") then
                if not OriginalStates.Parts[fx] then OriginalStates.Parts[fx]={_t="postfx",Enabled=fx.Enabled} end
                pcall(function() fx.Enabled=false end)
            elseif fx:IsA("Sky") then
                if not OriginalStates.Parts[fx] then OriginalStates.Parts[fx]={_t="sky",Parent=fx.Parent} end
                pcall(function() fx.Parent=OriginalStates.HiddenObjects end)
            end
        end
        SetQuality("low")
        local items=workspace:GetDescendants()
        for idx,obj in ipairs(items) do OtimizarObj(obj); if idx%180==0 then task.wait() end end
        descendantConn=workspace.DescendantAdded:Connect(OtimizarObj)
        Notify("FPS Booster V7","Limpeza concluída! "..optimizedCount.." objetos otimizados.",5,"ok")
    else
        if descendantConn then descendantConn:Disconnect(); descendantConn=nil end
        SetQuality("auto")
        if OriginalStates.Lighting.GlobalShadows~=nil then
            Lighting.GlobalShadows=OriginalStates.Lighting.GlobalShadows; Lighting.Brightness=OriginalStates.Lighting.Brightness
            Lighting.FogEnd=OriginalStates.Lighting.FogEnd; workspace.Terrain.WaterWaveSize=OriginalStates.Lighting.WaterWaveSize
            workspace.Terrain.WaterWaveSpeed=OriginalStates.Lighting.WaterWaveSpeed
            workspace.Terrain.WaterReflectance=OriginalStates.Lighting.WaterReflectance
        end
        local snap={}; for obj,props in pairs(OriginalStates.Parts) do snap[#snap+1]={obj=obj,props=props} end
        for idx,entry in ipairs(snap) do
            local obj,props=entry.obj,entry.props
            pcall(function()
                if typeof(obj)~="Instance" then return end
                local t=props._t
                if t=="model" or t=="pbr" or t=="sky" then
                    if props.Parent then obj.Parent=props.Parent end
                elseif t=="decal" then obj.Transparency=props.Transparency
                elseif t=="emitter" then obj.Enabled=props.Enabled
                elseif t=="light" then obj.Enabled=props.Enabled; obj.Brightness=props.Brightness
                elseif t=="prompt" then obj.MaxActivationDistance=props.MaxActivationDistance
                elseif t=="postfx" then obj.Enabled=props.Enabled
                elseif t=="union" or t=="part" then
                    obj.Material=props.Material; obj.CastShadow=props.CastShadow; obj.Reflectance=props.Reflectance
                    if props.RenderFidelity then obj.RenderFidelity=props.RenderFidelity end
                end
            end)
            if idx%180==0 then task.wait() end
        end
        for _,obj in ipairs(OriginalStates.HiddenObjects:GetChildren()) do
            pcall(function() local p=OriginalStates.Parts[obj]; if p and p.Parent then obj.Parent=p.Parent end end)
        end
        OriginalStates.Parts={}; OriginalStates.Lighting={}; setmetatable(OriginalStates.Parts,{__mode="k"})
        optimizedCount=0; Notify("FPS Booster V7","Gráficos restaurados completamente.",4,"ok")
    end
end

local function VisThroughWalls(part)
    if not OmegaState.Aimbot.WallCheck then return true end
    local cam = workspace.CurrentCamera
    if not cam then return true end
    local ray = Ray.new(cam.CFrame.Position, (part.Position - cam.CFrame.Position).Unit * 1000)
    return workspace:FindPartOnRayWithIgnoreList(ray, {Player.Character, part.Parent}) == nil
end

local function GetAimbotTargetPart(char)
    if not char then return nil end
    if OmegaState.Aimbot.TargetPart == "Chest" then
        return char:FindFirstChild("UpperTorso") or char:FindFirstChild("Torso") or char:FindFirstChild("HumanoidRootPart")
    end
    return char:FindFirstChild("Head") or char:FindFirstChild("UpperTorso") or char:FindFirstChild("Torso") or char:FindFirstChild("HumanoidRootPart")
end

local function BestTarget()
    local cam = workspace.CurrentCamera
    if not cam then return nil end
    local center = Vector2.new(cam.ViewportSize.X/2, cam.ViewportSize.Y/2)
    local chosen, shortest = nil, OmegaState.FOV.Radius

    for _, plr in ipairs(Players:GetPlayers()) do
        if plr ~= Player and plr.Character and plr.Character:FindFirstChild("Humanoid") and plr.Character.Humanoid.Health > 0 then
            if OmegaState.Aimbot.TeamCheck and plr.Team == Player.Team then continue end
            local bone = GetAimbotTargetPart(plr.Character)
            if bone and VisThroughWalls(bone) then
                local sp, on = cam:WorldToViewportPoint(bone.Position)
                if on then
                    local dist = (Vector2.new(sp.X, sp.Y) - center).Magnitude
                    if dist <= shortest then
                        shortest = dist
                        chosen = bone
                    end
                end
            end
        end
    end
    return chosen
end

-- AIMLOCK
RunService.RenderStepped:Connect(function()
    local cam = workspace.CurrentCamera
    if not OmegaState.Aimbot.Enabled or not cam then return end

    local tgt = BestTarget()
    OmegaState.Aimbot.CurrentTarget = tgt

    if tgt then
        local aimPos = tgt.Position

        if tgt.Name == "Head" then
            aimPos = tgt.Position + Vector3.new(0, -0.6, 0)
        elseif tgt.Name == "UpperTorso" or tgt.Name == "Torso" then
            aimPos = tgt.Position + Vector3.new(0, 0.15, 0)
        end

        if OmegaState.Aimbot.Prediction then
            local hrp = tgt.Parent:FindFirstChild("HumanoidRootPart")
            if hrp then
                aimPos = aimPos + (hrp.AssemblyLinearVelocity * OmegaState.Aimbot.PredictionFactor)
            end
        end

        if OmegaState.Aimbot.AimLock then
            cam.CFrame = CFrame.new(cam.CFrame.Position, aimPos)
        end
    end
end)

                    end
                end
            end

    for _,d in ipairs(OmegaState.Vector.parkourData) do table.insert(all,d) end

            table.insert(OmegaState.Vector.Pool,part)
